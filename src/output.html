func (m *Macchiato) renderListItem(item ListItem) string {
    if item.Type == Ordered {
        return fmt.Sprintf("<li value=\"%d\">%s</li>\n", item.Number, m.parseLine(item.Content))
    }
    return fmt.Sprintf("<li>%s</li>\n", m.parseLine(item.Content))
}
